project(GE_lab)
cmake_minimum_required(VERSION 3.11)
# Require modern C++
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

#### Setup Directories ####
# Main output directory
SET(OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin/")
# Ouput all DLLs from all libs into main build folder
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${OUTPUT_DIRECTORY})

#### Add External Dependencies ####
add_subdirectory("lib/SFML")
set(SFML_INCS "lib/SFML/include")
link_directories("${CMAKE_BINARY_DIR}/lib/SFML/lib")
#### level system ####
add_library(tile_level STATIC "lib/tile_level_loader/level_system.cpp")
target_include_directories(tile_level INTERFACE "lib/tile_level_loader")
target_link_libraries(tile_level sfml-graphics)

add_library(engine STATIC "lib/engine/game_system.cpp" "lib/engine/Renderer.cpp" "lib/engine/ecm.cpp")
target_include_directories(engine INTERFACE "lib/engine")
target_link_libraries(engine sfml-graphics)

#### Practical 04 ####
add_executable(lab_05 main.cpp Scenes.cpp Player.cpp Ghost.cpp components.cpp)
target_include_directories(lab_05 PRIVATE ${SFML_INCS})
target_link_libraries(lab_05 sfml-graphics tile_level engine)

set_target_properties(lab_05 
    PROPERTIES VS_DEBUGGER_WORKING_DIRECTORY
    ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/$(Configuration)
)
add_custom_target(copy_resources ALL COMMAND ${CMAKE_COMMAND} 
  -E copy_directory
    "${PROJECT_SOURCE_DIR}/res"
    ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/$<CONFIGURATION>/res
)
add_dependencies(lab_05 copy_resources)
